# Specify the minimum version of CMake required. 3.10 is a safe choice.
cmake_minimum_required(VERSION 3.10)

# Define the project name and specify that it's a C project.
project(tips VERSION 1.0.2 LANGUAGES C)

# Define the executable
add_executable(tips src/main.c)

# Handle the Data Directory
set(TIPS_DATA_DIR "${CMAKE_INSTALL_PREFIX}/share/tips/tips-pages" CACHE PATH "Path to tips pages files")

# Create a macro to store the path
target_compile_definitions(tips PRIVATE "-DTIPS_DATA_D=\"${TIPS_PAGES_PATH}\"")

# Install tips
install(TARGETS tips DESTINATION bin)
